# ============================================================================
# Data Source Manager
# ============================================================================

class DataSourceManager:
    """
    Manages data source connections with template handlers.
    
    This class provides a unified interface for:
    - Registering and managing data sources
    - Testing connections
    - Discovering schemas
    - Promoting project sources to global registry
    
    Attributes:
        handlers: Dictionary mapping source types to handlers
        store: ControlPanelStore for persistence
        config: AgenticConfig instance
    """
    
    def __init__(self, config: Optional[AgenticConfig] = None):
        """Initialize the data source manager."""
        self.config = config or AgenticConfig()
        self.store = ControlPanelStore.get_instance(self.config)
        
        # Register default handlers
        self.handlers: Dict[str, DataSourceHandler] = {}
        self.register_handler(DatabaseHandler())
        self.register_handler(FileStoreHandler())
        self.register_handler(APIHandler())
        
        logger.info("DataSourceManager initialized with handlers: " + 
                   ", ".join(self.handlers.keys()))
    
    def register_handler(self, handler: DataSourceHandler) -> None:
        """
        Register a data source handler.
        
        Args:
            handler: Handler instance to register
        """
        self.handlers[handler.source_type] = handler
        logger.debug(f"Registered handler for: {handler.source_type}")
    
    def get_handler(self, source_type: str) -> Optional[DataSourceHandler]:
        """Get handler for a source type."""
        return self.handlers.get(source_type)
