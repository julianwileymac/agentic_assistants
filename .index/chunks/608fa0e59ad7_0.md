"""
Dataset type system for the Agentic Data Catalog.

This module provides Kedro-inspired dataset abstractions for loading and saving
data from various sources with versioning support.

Example:
    >>> from agentic_assistants.data.datasets import CSVDataset, ParquetDataset
    >>> 
    >>> # Load a CSV file
    >>> dataset = CSVDataset(filepath="data/users.csv")
    >>> df = dataset.load()
    >>> 
    >>> # Save with versioning
    >>> versioned = ParquetDataset(filepath="data/output.parquet", versioned=True)
    >>> versioned.save(df)
    >>> 
    >>> # Load from URL
    >>> from agentic_assistants.data.datasets import URLDataset
    >>> url_data = URLDataset(url="https://api.example.com/data.json")
    >>> data = url_data.load()
    >>> 
    >>> # Scrape a website
    >>> from agentic_assistants.data.datasets import WebsiteDataset
    >>> website = WebsiteDataset(
    ...     url="https://example.com",
    ...     selectors={"title": "h1", "content": "article"}
    ... )
    >>> content = website.load()
"""
