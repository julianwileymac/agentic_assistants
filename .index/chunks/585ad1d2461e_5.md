    def has_updates(self) -> bool:
        """
        Check if the feed has new entries since last fetch.
        
        This does a lightweight check without fully parsing.
        """
        if not self._last_entry_ids:
            return True
        
        try:
            parsed = self._fetch_feed()
            current_ids = {
                e.get("id") or e.get("link") or e.get("title", "")
                for e in parsed.entries
            }
            return bool(current_ids - self._last_entry_ids)
        except Exception:
            return True
