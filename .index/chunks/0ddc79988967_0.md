"""
Augmenters for enriching text content during RAG ingestion.

Augmenters add additional metadata and content to chunks,
improving retrieval quality and providing richer context.
"""

import hashlib
import re

logger = get_logger(__name__)


class Augmenter(ABC):
    """Abstract base class for content augmenters."""
    
    name: str = "base"
    
    @abstractmethod
    def augment(self, chunk: Chunk, context: Dict[str, Any]) -> Chunk:
        """
        Augment a chunk with additional data.
        
        Args:
            chunk: The chunk to augment
            context: Additional context (source URL, document info, etc.)
            
        Returns:
            Augmented chunk (may be modified in place)
        """
        pass
    
    def augment_batch(
        self,
        chunks: List[Chunk],
        context: Dict[str, Any],
    ) -> List[Chunk]:
        """Augment multiple chunks."""
        return [self.augment(chunk, context) for chunk in chunks]


class MetadataAugmenter(Augmenter):
    """
    Add standard metadata to chunks.
    
    Adds:
    - Source URL/path
    - Timestamp
    - Content hash
    - Word count
    - Character count
    """
    
    name = "metadata"
    
    def __init__(
        self,
        include_hash: bool = True,
        include_stats: bool = True,
        custom_fields: Optional[Dict[str, Any]] = None,
    ):
        """
        Initialize metadata augmenter.
        
        Args:
            include_hash: Include content hash
            include_stats: Include word/char counts
            custom_fields: Additional fields to add to all chunks
        """
        self.include_hash = include_hash
        self.include_stats = include_stats
        self.custom_fields = custom_fields or {}
    
    def augment(self, chunk: Chunk, context: Dict[str, Any]) -> Chunk:
        # Add source info

            chunk.metadata["source_type"] = context["source_type"]

        chunk.metadata["ingested_at"] = datetime.utcnow().isoformat()
