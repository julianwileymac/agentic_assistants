    """Add or remove tags from an artifact."""
    logger.info(f"Updating tags for artifact: {artifact_id}")
    
    storage = get_storage()
    artifact = storage.get(artifact_id)
    
    if not artifact:
        raise HTTPException(status_code=404, detail="Artifact not found")
