    def get_session(self, name: str) -> Optional[Session]:
        """
        Get a session by name.
        
        Args:

        Returns:
            Session instance or None if not found
        """
        with self._lock:

            with self._get_connection() as conn:
                row = conn.execute(
                    "SELECT id, name, created_at FROM sessions WHERE name = ?",
                    (name,),
                ).fetchone()
                
                if row is None:
                    return None
