Warm L1 cache from L2.
        
        Args:
            keys: Keys to warm
            
        Returns:
            Number of keys warmed
        """
        if not self._l2_available or not self._l2:
            return 0
        
        count = 0
        
        for key in keys:
            if not self._l1.exists(key):
                value = self._l2.get(key)
                if value is not None:
                    self._l1.set(key, value, self.config.l1_ttl_seconds)
                    count += 1
        
        return count
