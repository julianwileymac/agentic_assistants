    # e.g. 's' b''
    message = "cannot mix bytes and nonbytes literals"

    def _is_bytes_literal(self, string):
        if string.type == 'fstring':
            return False

    def is_issue(self, node):
        first = node.children[0]

            if first_is_bytes != self._is_bytes_literal(string):
