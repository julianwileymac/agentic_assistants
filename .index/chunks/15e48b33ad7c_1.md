class SharedStorageManager:
    """
    Manages shared storage for artifacts across sessions.
    
    Features:
    - Copy artifacts to shared storage
    - Organize by tags and groups
    - Cross-session artifact access
    - Automatic symlink creation in tag/group directories
    """
    
    def __init__(self, config: Optional[AgenticConfig] = None):
        self.config = config or AgenticConfig()
        self.data_layer = DataLayer(config)
        
        # Set up directory structure
        self.shared_dir = Path(self.config.data_dir) / "shared" / "artifacts"
        self.by_tag_dir = self.shared_dir / "by-tag"
        self.by_group_dir = self.shared_dir / "by-group"
        self.index_file = self.shared_dir / "index.json"
        
        self._ensure_directories()
        self._load_index()
    
    def _ensure_directories(self):
        """Ensure all required directories exist."""
        self.shared_dir.mkdir(parents=True, exist_ok=True)
        self.by_tag_dir.mkdir(parents=True, exist_ok=True)
        self.by_group_dir.mkdir(parents=True, exist_ok=True)
    
    def _load_index(self):
        """Load the artifact index."""
        self._index: dict[str, SharedArtifactMetadata] = {}
        
        if self.index_file.exists():
            try:
                with open(self.index_file, "r") as f:
                    data = json.load(f)
                    for artifact_id, meta in data.items():
                        self._index[artifact_id] = SharedArtifactMetadata.from_dict(meta)
            except Exception as e:
                logger.error(f"Failed to load shared storage index: {e}")
    
    def _save_index(self):
        """Save the artifact index."""
        try:
            data = {
                artifact_id: meta.to_dict()
                for artifact_id, meta in self._index.items()
            }
            with open(self.index_file, "w") as f:
                json.dump(data, f, indent=2)
        except Exception as e:
            logger.error(f"Failed to save shared storage index: {e}")
