class QualityAnnotator(Annotator):
    """
    Assess quality/relevance of chunk content.
    
    Useful for filtering out low-quality content.
    """
    
    name = "quality"
    
    def __init__(
        self,
        min_word_count: int = 10,
        check_repetition: bool = True,
        use_llm: bool = False,
        llm_fn: Optional[Callable] = None,
    ):
        """
        Initialize quality annotator.
        
        Args:
            min_word_count: Minimum words for good quality
            check_repetition: Check for repetitive content
            use_llm: Use LLM for quality assessment
            llm_fn: Custom LLM function
        """
        self.min_word_count = min_word_count
        self.check_repetition = check_repetition
        self.use_llm = use_llm
        self.llm_fn = llm_fn
    
    def _heuristic_quality(self, text: str) -> Tuple[str, float]:
        """Assess quality using heuristics."""
        score = 1.0
        issues = []
        
        words = text.split()
        word_count = len(words)
        
        # Check word count
        if word_count < self.min_word_count:
            score -= 0.3
            issues.append("too_short")
        
        # Check for repetition
        if self.check_repetition and word_count > 0:
            unique_words = len(set(w.lower() for w in words))
            uniqueness = unique_words / word_count
            if uniqueness < 0.3:
                score -= 0.3
                issues.append("repetitive")
        
        # Check for meaningful content (not just numbers/symbols)
        alpha_ratio = sum(c.isalpha() for c in text) / len(text) if text else 0
        if alpha_ratio < 0.5:
            score -= 0.2
            issues.append("low_text_ratio")
        
        # Determine label
        if score >= 0.8:
            label = "high"
        elif score >= 0.5:
            label = "medium"
        else:
            label = "low"
        
        return label, max(0.0, score), issues
    
    def annotate(self, chunk: Chunk, context: Dict[str, Any]) -> Chunk:
        label, score, issues = self._heuristic_quality(chunk.content)
        
        chunk.metadata["quality"] = label
        chunk.metadata["quality_score"] = round(score, 3)
        if issues:
            chunk.metadata["quality_issues"] = issues
        
        return chunk
