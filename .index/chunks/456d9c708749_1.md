    def _configure_credentials(self) -> None:
        """Configure Kaggle API credentials."""
        if self._credentials:
            os.environ["KAGGLE_USERNAME"] = self._credentials.get("username", "")

    def _get_kaggle_api(self):
        """Get authenticated Kaggle API instance."""
        try:
            from kaggle.api.kaggle_api_extended import KaggleApi
            
            api = KaggleApi()
            api.authenticate()
            return api
        except ImportError:
            raise DatasetError(
                "kaggle library is required. Install with: pip install kaggle"
            )
        except Exception as e:
            raise DatasetError(
                f"Failed to authenticate with Kaggle API: {e}. "
                "Ensure credentials are in ~/.kaggle/kaggle.json or "
                "set KAGGLE_USERNAME and KAGGLE_KEY environment variables."
            )
    
    def _get_download_path(self) -> Path:
