    >>> 
    >>> # Read a public CSV file
    >>> dataset = S3PublicDataset(
    ...     bucket="my-public-bucket",
    ...     key="data/file.csv"
    ... )
    >>> df = dataset.load()
    >>> 
    >>> # List files in a bucket prefix
    >>> dataset = S3PublicDataset(bucket="aws-public-data", prefix="common-crawl/")
    >>> files = dataset.list_objects()
