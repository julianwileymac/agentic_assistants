"""
Data tagging system for training datasets.

This module provides a comprehensive tagging system for managing
metadata and labels on training data.
"""

import json
import uuid
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from pathlib import Path
from typing import Any, Dict, List, Optional, Set

from agentic_assistants.utils.logging import get_logger

logger = get_logger(__name__)


class TagCategory(str, Enum):
    """Categories for data tags."""
    
    # Data type tags
    DATA_TYPE = "data_type"         # instruct, preference, completion, etc.
    
    # Quality tags
    QUALITY = "quality"             # high_quality, needs_review, verified, etc.
    
    # Domain tags
    DOMAIN = "domain"               # code, medical, legal, general, etc.
    
    # Task tags
    TASK = "task"                   # summarization, qa, chat, reasoning, etc.
    
    # Source tags
    SOURCE = "source"               # synthetic, human, crawled, etc.
    
    # Processing tags
    PROCESSING = "processing"       # raw, cleaned, augmented, filtered, etc.
    
    # Language tags
    LANGUAGE = "language"           # en, es, zh, multilingual, etc.
    
    # Custom tags
    CUSTOM = "custom"


@dataclass
class DataTag:
    """A tag that can be applied to datasets or data items."""
    
    id: str
    name: str
    category: TagCategory
    description: str = ""
    color: str = "#808080"  # Hex color for UI
    
    # Metadata
    created_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())
    created_by: Optional[str] = None
    
    # Usage tracking
    usage_count: int = 0
    
    def to_dict(self) -> Dict[str, Any]:
        return {
            "id": self.id,
            "name": self.name,
            "category": self.category.value,
            "description": self.description,
            "color": self.color,
            "created_at": self.created_at,
            "created_by": self.created_by,
            "usage_count": self.usage_count,
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "DataTag":
        return cls(
            id=data["id"],
            name=data["name"],
            category=TagCategory(data.get("category", "custom")),
            description=data.get("description", ""),
            color=data.get("color", "#808080"),
            created_at=data.get("created_at", datetime.utcnow().isoformat()),
            created_by=data.get("created_by"),
            usage_count=data.get("usage_count", 0),
        )
