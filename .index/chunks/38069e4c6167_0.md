class LocalCache(CacheBackend):
    """
    Thread-safe local LRU cache with disk spillover.
    
    Features:
    - LRU eviction policy
    - TTL support
    - Disk spillover for persistence
    - Thread-safe operations
    
    Example:
        >>> cache = LocalCache(CacheConfig(max_size=100))
        >>> cache.set("key", {"data": "value"}, ttl_seconds=3600)
        >>> value = cache.get("key")
    """
    
    def __init__(self, config: Optional[CacheConfig] = None):
        """
        Initialize local cache.
        
        Args:
            config: Cache configuration
        """
        config = config or CacheConfig(backend="local")
        super().__init__(config)
        
        self._cache: OrderedDict[str, CacheEntry] = OrderedDict()
        self._lock = threading.RLock()
        
        # Disk spillover
        if self.config.disk_spillover:
            self._disk_path = Path(self.config.disk_path)
            self._disk_path.mkdir(parents=True, exist_ok=True)
    
    def _evict_if_needed(self) -> None:
        """Evict entries if cache is full."""
        while len(self._cache) >= self.config.max_size:
            # Remove oldest entry
            key, _ = self._cache.popitem(last=False)
            self._stats.evictions += 1
            logger.debug(f"Evicted cache entry: {key}")
    
    def _prune_expired(self) -> int:
        """Remove expired entries."""
        expired_keys = []
        
        for key, entry in self._cache.items():
            if entry.is_expired:
                expired_keys.append(key)
        
        for key in expired_keys:
            del self._cache[key]
            self._stats.expirations += 1
        
        return len(expired_keys)
    
    def _load_from_disk(self, key: str) -> Optional[CacheEntry]:
        """Load entry from disk spillover."""
        if not self.config.disk_spillover:
            return None

        file_path = self._disk_path / f"{key_hash}.cache"
        
        if not file_path.exists():
            return None
        
        try:
            with open(file_path, "rb") as f:
                data = pickle.load(f)
            
            entry = CacheEntry.from_dict(data)
            
            # Check if expired
            if entry.is_expired:
                file_path.unlink()
                return None
