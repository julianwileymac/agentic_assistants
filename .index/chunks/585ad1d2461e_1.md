class RSSDataset(AbstractDataset[Dict[str, Any]]):
    """
    Dataset for fetching and parsing RSS/Atom feeds.
    
    This dataset supports:
    - RSS 1.0, RSS 2.0, and Atom feeds
    - Entry filtering by date
    - Content extraction from entries
    - Enclosure/media handling
    - Feed metadata extraction
    - Change detection for monitoring
    
    Attributes:
        url: Feed URL
        max_entries: Maximum entries to return
        since_hours: Only return entries from last N hours
        
    Example:
        >>> # Basic feed fetch
        >>> dataset = RSSDataset(url="https://example.com/feed")
        >>> result = dataset.load()
        >>> for entry in result["entries"]:
        ...     print(entry["title"])
        
        >>> # Monitor for new entries
        >>> dataset = RSSDataset(url="https://news.site.com/rss")
        >>> new_entries = dataset.get_new_entries()
    """
    
    DEFAULT_LOAD_ARGS: Dict[str, Any] = {
        "timeout": 30,
        "include_content": True,
        "sanitize_html": True,
    }
    
    def __init__(
        self,
        url: str,
        max_entries: Optional[int] = None,
        since_hours: Optional[float] = None,
        since_date: Optional[datetime] = None,
        load_args: Optional[Dict[str, Any]] = None,
        save_args: Optional[Dict[str, Any]] = None,
        credentials: Optional[Dict[str, Any]] = None,
        cache_dir: Optional[str] = None,
        metadata: Optional[DatasetMetadata] = None,
        **kwargs,
    ):
        """
        Initialize the RSS dataset.
        
        Args:
            url: Feed URL
            max_entries: Maximum number of entries to return
            since_hours: Only return entries from the last N hours
            since_date: Only return entries after this date
            load_args: Loading arguments
            save_args: Saving arguments
            credentials: Credentials for authenticated feeds
            cache_dir: Directory for caching
            metadata: Dataset metadata
        """
        super().__init__(
            filepath=url,
            load_args=load_args,
            save_args=save_args,
            credentials=credentials,
            metadata=metadata,
        )
        
        self.url = url
        self.max_entries = max_entries
        self.since_hours = since_hours
        self.since_date = since_date
        self.cache_dir = Path(cache_dir) if cache_dir else Path("./data/rss_cache")
        
        # State for change detection
        self._last_fetch: Optional[datetime] = None
        self._last_entry_ids: set = set()
        self._feed_info: Optional[FeedInfo] = None
    
    def _get_feedparser(self):
        """Get feedparser module."""
        try:
            import feedparser
            return feedparser
