class MutexValue:

    _multiprocess = False

def MultiProcessValue(process_identifier=os.getpid):

    """
    files = {}
    values = []
    pid = {'value': process_identifier()}

    # as we presume this means there is no threading going on.
    # This avoids the need to also have mutexes in __MmapDict.
