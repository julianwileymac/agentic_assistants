def test_datetime_tzinfo():
    tz = pytz.timezone('Australia/Sydney')
    dt = datetime.datetime(2002, 2, 20, 13, 37, 42, 7, tzinfo=tz)
    w = DatetimePicker(value=dt)
    assert w.value == dt
    # tzinfo only changes upon input from user
    assert w.value.tzinfo == tz
