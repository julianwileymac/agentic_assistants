// ============================================================================
// Recent Activity Store
// ============================================================================

interface Activity {
  id: string;
  type: 'project' | 'agent' | 'flow' | 'component' | 'experiment';
  action: 'create' | 'update' | 'delete' | 'run' | 'view';
  resourceId: string;
  resourceName: string;
  timestamp: Date;
}

interface ActivityStore {
  activities: Activity[];
  maxActivities: number;

  addActivity: (activity: Omit<Activity, 'id' | 'timestamp'>) => void;
  clearActivities: () => void;
}

export const useActivityStore = create<ActivityStore>()(
  persist(
    (set) => ({
      activities: [],
      maxActivities: 50,

      addActivity: (activity) =>
        set((state) => {
          const newActivity: Activity = {
            ...activity,
            id: Math.random().toString(36).substring(7),
            timestamp: new Date(),
          };
          const activities = [newActivity, ...state.activities].slice(
            0,
            state.maxActivities
          );
          return { activities };
        }),

      clearActivities: () => set({ activities: [] }),
    }),
    {
      name: 'activity-storage',
      partialize: (state) => ({
        activities: state.activities.slice(0, 20), // Only persist last 20
      }),
    }
  )
);
