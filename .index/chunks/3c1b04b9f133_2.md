    def _load_data(self) -> None:
        """Load lineage data from storage."""
        lineage_file = self.storage_path / "lineage.json"
        relations_file = self.storage_path / "relations.json"
        
        if lineage_file.exists():
            try:
                with open(lineage_file, "r") as f:
                    data = json.load(f)
                for item in data.get("lineage", []):
                    lineage = TrainingDataLineage.from_dict(item)
                    self._lineage[lineage.id] = lineage
            except Exception as e:
                logger.warning(f"Failed to load lineage: {e}")
        
        if relations_file.exists():
            try:
                with open(relations_file, "r") as f:
                    data = json.load(f)
                for item in data.get("relations", []):
                    relation = DataRelation(
                        id=item["id"],
                        source_id=item["source_id"],
                        source_type=item["source_type"],
                        target_id=item["target_id"],
                        target_type=item["target_type"],
                        relationship=LineageRelationship(item["relationship"]),
                        description=item.get("description"),
                        transformation=item.get("transformation"),
                        created_at=item.get("created_at", ""),
                        metadata=item.get("metadata", {}),
                    )
                    self._relations[relation.id] = relation
            except Exception as e:
                logger.warning(f"Failed to load relations: {e}")
    
    def _save_data(self) -> None:
        """Save lineage data to storage."""
        lineage_file = self.storage_path / "lineage.json"
        relations_file = self.storage_path / "relations.json"
        
        with open(lineage_file, "w") as f:
            json.dump({
                "lineage": [l.to_dict() for l in self._lineage.values()]
            }, f, indent=2)
        
        with open(relations_file, "w") as f:
            json.dump({
                "relations": [r.to_dict() for r in self._relations.values()]
