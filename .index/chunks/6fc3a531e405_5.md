ataset(AbstractDataset[T]):
    """
    Wrapper that adds versioning to any dataset.
    
    This class wraps another dataset type and adds automatic versioning
    with timestamp-based version directories.
    
    Example:
        >>> base_dataset = CSVDataset(filepath="data/users.csv")
        >>> versioned = VersionedDataset(base_dataset)
        >>> versioned.save(df)  # Saves to data/users/<timestamp>/users.csv
    """
    
    def __init__(
        self,
        dataset: AbstractDataset[T],
        version: Optional[Version] = None,
    ):
        """
        Initialize the versioned dataset wrapper.
        
        Args:
            dataset: The dataset to wrap
            version: Optional specific version
        """
        self._wrapped = dataset
        self._version = version or Version()
        
        # Enable versioning on wrapped dataset
        self._wrapped._versioned = True
        self._wrapped._version = self._version
    
    @property
    def filepath(self) -> str:
        return self._wrapped.filepath
    
    @property
    def versioned(self) -> bool:
        return True
    
    @property
    def metadata(self) -> DatasetMetadata:
        return self._wrapped.metadata
    
    def load(self) -> T:
        return self._wrapped.load()
    
    def save(self, data: T) -> None:
        self._wrapped.save(data)
    
    def exists(self) -> bool:
        return self._wrapped.exists()
    
    def release(self) -> None:
        self._wrapped.release()
    
    def _load(self) -> T:
        return self._wrapped._load()
    
    def _save(self, data: T) -> None:
        self._wrapped._save(data)
    
    def list_versions(self) -> list[str]:
        """List all available versions."""
        return self._wrapped._list_versions()
    
    def load_version(self, version: str) -> T:
        """Load a specific version."""
        self._wrapped._version = Version(load_version=version)
        return self._wrapped.load()
    
    def __repr__(self) -> str:
        return f"VersionedDataset({self._wrapped!r})"
