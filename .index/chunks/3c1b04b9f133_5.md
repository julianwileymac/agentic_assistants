        self,
        entity_id: str,
        max_depth: int = 10,
    ) -> List[DataRelation]:
        """
        Get downstream relations (dependents) for an entity.
        
        Args:
            entity_id: Entity ID
            max_depth: Maximum traversal depth
        
        Returns:
            List of downstream relations
        """
        downstream = []
        visited = set()
        queue = [(entity_id, 0)]
        
        while queue:
            current_id, depth = queue.pop(0)
            
            if current_id in visited or depth > max_depth:
                continue
            
            visited.add(current_id)
            
            for relation in self._relations.values():
                if relation.source_id == current_id:
                    downstream.append(relation)
                    queue.append((relation.target_id, depth + 1))
        
        return downstream
