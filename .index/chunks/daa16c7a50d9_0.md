"""
Hybrid multi-tier cache with local L1 and distributed L2.
"""

from typing import Any, Callable, Dict, List, Optional

from agentic_assistants.data.caching.base import (
    CacheBackend,
    CacheConfig,
    CacheStats,
)
from agentic_assistants.data.caching.local_cache import LocalCache
from agentic_assistants.data.caching.redis_cache import RedisCache
from agentic_assistants.utils.logging import get_logger

logger = get_logger(__name__)


class HybridCache(CacheBackend):
    """
    Multi-tier hybrid cache with local L1 and distributed L2.
    
    Architecture:
    - L1: Fast local LRU cache for hot data
    - L2: Distributed Redis cache for shared data
    
    Read flow:
    1. Check L1 (local) → return if found
    2. Check L2 (distributed) → promote to L1 and return if found
    3. Cache miss
    
    Write flow:
    1. Write to L1 (local)
    2. Write to L2 (distributed)
    
    Features:
    - Low latency for frequently accessed data
    - Distributed consistency for shared data
    - Automatic promotion from L2 to L1
    - Graceful degradation if L2 is unavailable
    
    Example:
        >>> config = CacheConfig(
        ...     backend="hybrid",
        ...     l1_max_size=100,
        ...     l1_ttl_seconds=300,
        ...     redis_url="redis://localhost:6379"
        ... )
        >>> cache = HybridCache(config)
        >>> cache.set("key", {"data": "value"})
    """
    
    def __init__(self, config: Optional[CacheConfig] = None):
        """
        Initialize hybrid cache.
        
        Args:
            config: Cache configuration
        """
        config = config or CacheConfig(backend="hybrid")
        super().__init__(config)
        
        # Create L1 local cache
        l1_config = CacheConfig(
            backend="local",
            max_size=config.l1_max_size,
            default_ttl_seconds=config.l1_ttl_seconds,
            disk_spillover=False,  # No disk for L1
        )
        self._l1 = LocalCache(l1_config)
        
        # Create L2 distributed cache
        l2_backend = config.l2_backend
        
        if l2_backend == "redis":
            try:
                self._l2 = RedisCache(config)
                self._l2_available = True
            except Exception as e:
                logger.warning(f"L2 Redis cache unavailable: {e}")
                self._l2 = None
                self._l2_available = False
        else:
            # Could add memcached or other backends here
            self._l2 = None
            self._l2_available = False
        
        # Track stats for both tiers
        self._l1_stats = CacheStats(max_size=config.l1_max_size)
        self._l2_stats = CacheStats(max_size=config.max_size)
    
    def get(self, key: str, default: Any = None) -> Optional[Any]:
        """Get a value from the cache."""
        # Check L1 first
        value = self._l1.get(key)
