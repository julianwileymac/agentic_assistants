        uri = params.get("uri", "")
        
        # Parse file:// URI

        try:
            from pathlib import Path
            path = Path(file_path)

                content = path.read_text(encoding="utf-8")

                    "contents": [
                        {
                            "uri": uri,

                    ],
                }
