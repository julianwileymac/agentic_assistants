        """
        Extending a validator preserves its notion of schema IDs.
        """

            return schema.get("__test__", self.Validator.ID_OF(schema))
        correct_id = "the://correct/id/"
        meta_schema = {
            "$id": "the://wrong/id/",

            meta_schema=meta_schema,
            validators=self.validators,
