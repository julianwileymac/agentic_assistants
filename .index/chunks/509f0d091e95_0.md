    >>> from langgraph.graph import StateGraph
    >>> from agentic_assistants.adapters import LangGraphAdapter
    >>> 
    >>> adapter = LangGraphAdapter()
    >>> 
    >>> # Create your graph
    >>> graph = StateGraph(MyState)
    >>> graph.add_node("process", process_fn)
    >>> ...
    >>> 
    >>> # Run with tracking
    >>> result = adapter.run_graph(
