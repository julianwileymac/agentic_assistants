    def load_from_yaml(self, path: Union[str, Path]) -> int:
        """
        Load templates from a YAML file.
        
        The YAML file should contain a list of template definitions or
        a dictionary mapping IDs to template configs.
        
        Args:
            path: Path to YAML file
            
        Returns:
            Number of templates loaded
        """
        path = Path(path)
        
        if not path.exists():
            logger.warning(f"Template file not found: {path}")
            return 0
        
        with open(path, "r") as f:
            data = yaml.safe_load(f) or {}
        
        count = 0
        
        # Handle list format
        if isinstance(data, list):
            for template_config in data:
                try:
                    self.register(template_config)
                    count += 1
                except Exception as e:
                    logger.error(f"Failed to load template: {e}")
        
        # Handle dict format
        elif isinstance(data, dict):
            # Check if it's a single template or multiple
            if "id" in data:
                # Single template
                try:
                    self.register(data)
                    count += 1
                except Exception as e:
                    logger.error(f"Failed to load template: {e}")
            else:
                # Multiple templates keyed by ID
                for template_id, template_config in data.items():
                    try:
                        if "id" not in template_config:
                            template_config["id"] = template_id
                        self.register(template_config)
                        count += 1
                    except Exception as e:
                        logger.error(f"Failed to load template {template_id}: {e}")
        
        logger.info(f"Loaded {count} templates from {path}")
        return count
    
    def load_from_directory(self, directory: Union[str, Path]) -> int:
        """
        Load templates from all YAML files in a directory.
        
        Args:
            directory: Directory containing YAML files
            
        Returns:
            Total number of templates loaded
        """
        directory = Path(directory)
        
        if not directory.exists():
            logger.warning(f"Template directory not found: {directory}")
            return 0
        
        count = 0
        for yaml_file in directory.glob("*.yaml"):
            count += self.load_from_yaml(yaml_file)
        for yml_file in directory.glob("*.yml"):
