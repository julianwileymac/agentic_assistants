class CSVDataset(AbstractDataset[pd.DataFrame]):
    """
    Dataset for CSV files using Pandas.
    
    Example:
        >>> dataset = CSVDataset(
        ...     filepath="data/users.csv",
        ...     load_args={"encoding": "utf-8", "sep": ","},
        ...     save_args={"index": False},
        ... )
        >>> df = dataset.load()
    """
    
    DEFAULT_LOAD_ARGS: Dict[str, Any] = {}
    DEFAULT_SAVE_ARGS: Dict[str, Any] = {"index": False}
    
    def _load(self) -> pd.DataFrame:
        """Load CSV file into a pandas DataFrame."""
        return pd.read_csv(self.filepath, **self._load_args)
    
    def _save(self, data: pd.DataFrame) -> None:
        """Save pandas DataFrame to CSV file."""
        data.to_csv(self._filepath, **self._save_args)
    
    def _update_metadata_on_load(self, data: pd.DataFrame) -> None:
        """Update metadata with schema information."""
        super()._update_metadata_on_load(data)
        
        # Extract schema from DataFrame
        self._metadata.schema = {
            "columns": list(data.columns),
            "dtypes": {col: str(dtype) for col, dtype in data.dtypes.items()},
        }
        self._metadata.row_count = len(data)
