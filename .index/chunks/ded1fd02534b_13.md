        return "Unexpected token %r. Expected one of: {%s}" % (str(error.token), expected)

        return str(error)

def _get_mangle(prefix, aliases, base_mangle=None):
    def mangle(s):
        if s in aliases:
            s = aliases[s]
        else:
            if s[0] == '_':
                s = '_%s__%s' % (prefix, s[1:])
