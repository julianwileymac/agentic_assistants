    @classmethod

        """
        Create dataset from S3 URI.
        
        Args:
            uri: S3 URI (s3://bucket/key)
            **kwargs: Additional arguments
            
        Returns:
            S3PublicDataset instance
        """
        parsed = urlparse(uri)
        
        if parsed.scheme != "s3":
            raise DatasetError(f"Invalid S3 URI: {uri}")
        
        bucket = parsed.netloc
        key = parsed.path.lstrip("/")
        
        return cls(bucket=bucket, key=key, **kwargs)
