    """
    Format tokens as appropriate for a new testcase.

    .. versionadded:: 2.0
    """
    name = 'Testcase'
    aliases = ['testcase']

    def __init__(self, **options):

        if self.encoding is not None and self.encoding != 'utf-8':

    def format(self, tokensource, outfile):
        indentation = ' ' * 12

        outbuf = []
