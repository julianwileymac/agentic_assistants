    def list_layer_datasets(
        self,
        layer: Union[DataLayer, str],
        pattern: str = "*.*",
    ) -> List[str]:
        """
        List all datasets in a layer.
        
        Args:
            layer: DataLayer to list
            pattern: Glob pattern for filtering
            
        Returns:
            List of dataset names (without extension)
        """
        layer_path = self.get_layer_path(layer)
        
        if not layer_path.exists():
            return []
        
        datasets = []
        for file_path in layer_path.glob(pattern):
            if file_path.is_file() and not file_path.name.startswith("."):
                datasets.append(file_path.stem)
        
        return sorted(set(datasets))
