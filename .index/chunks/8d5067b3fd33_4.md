class LanguageAnnotator(Annotator):
    """
    Detect language of chunk content.
    """
    
    name = "language"
    
    def __init__(self, confidence_threshold: float = 0.8):
        """
        Initialize language annotator.
        
        Args:
            confidence_threshold: Minimum confidence for detection
        """
        self.confidence_threshold = confidence_threshold
        self._detector = None
    
    def _get_detector(self):
        """Get language detector."""
        if self._detector is not None:
            return self._detector
        
        try:
            from langdetect import detect_langs
            self._detector = detect_langs
            return self._detector
        except ImportError:
            return None
    
    def _heuristic_detect(self, text: str) -> Tuple[str, float]:
        """Simple heuristic language detection."""
        # Check for common language patterns
        
        # English indicators
        english_words = {"the", "is", "are", "and", "or", "to", "in", "for"}
        words = set(text.lower().split())
        english_overlap = len(words & english_words)
        
        if english_overlap >= 3:
            return "en", 0.8
        
        # Check for non-ASCII indicating other languages
        non_ascii = sum(1 for c in text if ord(c) > 127)
        if non_ascii / len(text) > 0.3:
            # Could be Asian, Cyrillic, etc.
            return "unknown", 0.3
        
        return "en", 0.5
    
    def annotate(self, chunk: Chunk, context: Dict[str, Any]) -> Chunk:
        detector = self._get_detector()
        
        if detector:
            try:
                results = detector(chunk.content[:1000])
                if results:
                    top_result = results[0]
                    chunk.metadata["language"] = top_result.lang
                    chunk.metadata["language_confidence"] = round(top_result.prob, 3)
                else:
                    chunk.metadata["language"] = "unknown"
                    chunk.metadata["language_confidence"] = 0.0
            except Exception:
                lang, conf = self._heuristic_detect(chunk.content)
                chunk.metadata["language"] = lang
                chunk.metadata["language_confidence"] = conf
        else:
            lang, conf = self._heuristic_detect(chunk.content)
            chunk.metadata["language"] = lang
            chunk.metadata["language_confidence"] = conf
        
        return chunk
