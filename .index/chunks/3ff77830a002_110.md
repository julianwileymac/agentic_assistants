class AbstractNextTerminalAfterProductionWalker extends RestWalker {
    constructor(topRule, occurrence) {

        this.occurrence = occurrence;

            token: undefined,
            occurrence: undefined,
            isEndOfRule: undefined,
        };
    }

}
