"""
Abstract base class for all dataset types.

This module provides the foundational dataset abstractions inspired by Kedro's
dataset system, with support for versioning, credentials, and filesystem abstraction.

Example:
    >>> from agentic_assistants.data.datasets.base import AbstractDataset
    >>> 
    >>> class MyDataset(AbstractDataset):
    ...     def _load(self):
    ...         return self._fs.open(self._filepath).read()
    ...     
    ...     def _save(self, data):
    ...         with self._fs.open(self._filepath, 'w') as f:
    ...             f.write(data)
"""

@dataclass
class Version:
    """
    Represents a dataset version.
    
    Attributes:
        load_version: Version to load (None for latest)
        save_version: Version to save (None for auto-generate)
    """
    load_version: Optional[str] = None
    save_version: Optional[str] = None
    
    @staticmethod
    def generate_timestamp() -> str:
        """Generate a timestamp-based version string."""
        return datetime.utcnow().strftime("%Y-%m-%dT%H.%M.%S.%fZ")
    
    @staticmethod
    def parse_timestamp(version: str) -> datetime:
        """Parse a version string back to datetime."""
        return datetime.strptime(version, "%Y-%m-%dT%H.%M.%S.%fZ")
