    def confirm_dataset_types(self, dataset_names: List[str]) -> List[str]:
        """
        Confirm which dataset names can be resolved.
        
        Args:
            dataset_names: List of dataset names to check
            
        Returns:
            List of dataset names that could not be resolved

    def create_from_template(
        self,
        name: str,
        template_id: str,
        params: Optional[Dict[str, Any]] = None,
        add_to_catalog: bool = True,
    ) -> AbstractDataset:
        """
        Create a dataset from a template.
        
        Templates provide pre-configured setups for common data sources.
        
        Args:
            name: Name for the dataset in the catalog
            template_id: ID of the template to use
            params: Parameters for the template
            add_to_catalog: Whether to add the dataset to the catalog
            
        Returns:
            Created dataset instance
            
        Example:
            >>> dataset = catalog.create_from_template(
            ...     name="github_repos",
            ...     template_id="github-api",
            ...     params={"endpoint": "/users/octocat/repos"}
            ... )
        """
        from agentic_assistants.data.templates import get_template
        
        template = get_template(template_id)
        dataset = template.create(**(params or {}))
        
        if add_to_catalog:
            self.add_dataset(name, dataset)
            logger.info(f"Created dataset '{name}' from template '{template_id}'")
        
        return dataset
