"""
Website Dataset for web scraping with configurable selectors.

This module provides a dataset implementation for scraping websites
with support for CSS selectors, XPath, and structured data extraction.

Example:
    >>> from agentic_assistants.data.datasets import WebsiteDataset
    >>> 
    >>> # Scrape article content
    >>> dataset = WebsiteDataset(
    ...     url="https://blog.example.com/article",
    ...     selectors={
    ...         "title": "h1.article-title",
    ...         "content": "div.article-body",
    ...         "author": "span.author-name",
    ...     }
    ... )
    >>> article = dataset.load()
"""

@dataclass
class SelectorConfig:
    """Configuration for a single selector."""
    
    selector: str
    selector_type: str = "css"  # css, xpath
    attribute: Optional[str] = None  # Extract attribute instead of text
    multiple: bool = False  # Return list of matches
    transform: Optional[str] = None  # strip, lower, upper, int, float
    default: Any = None  # Default value if not found
    
    @classmethod
    def from_string(cls, selector: str) -> "SelectorConfig":
        """Create config from simple selector string."""
        return cls(selector=selector)
    
    @classmethod
    def from_dict(cls, config: Dict[str, Any]) -> "SelectorConfig":
        """Create config from dictionary."""
        return cls(**config)


@dataclass
class ScrapedPage:
    """Result of a web scraping operation."""
    
    url: str
    data: Dict[str, Any]
    html: str
    scraped_at: datetime = field(default_factory=datetime.utcnow)
    status_code: int = 200
    elapsed_ms: float = 0.0
    links: List[str] = field(default_factory=list)
    
    @property
    def content_hash(self) -> str:
        """Compute hash of HTML for change detection."""
        return hashlib.sha256(self.html.encode()).hexdigest()


class WebsiteDataset(AbstractDataset[Dict[str, Any]]):
    """
