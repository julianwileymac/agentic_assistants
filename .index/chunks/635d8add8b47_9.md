export function useLinkProjectResource(projectId: string) {
  return useSWRMutation(
    `/api/v1/projects/${projectId}/resources/link`,
    async (url: string, { arg }: { arg: { resource_type: string; resource_id: string; alias?: string } }) => {
      const params = new URLSearchParams({
        resource_type: arg.resource_type,
        resource_id: arg.resource_id,
      });
      if (arg.alias) params.set('alias', arg.alias);
      return apiFetch<{ status: string }>(`${url}?${params.toString()}`, { method: 'POST' });
    }
  );
}
