class GitHubAPITemplate(DataSourceTemplate):
    """Template for GitHub REST API access."""
    
    id = "github-api"
    name = "GitHub API"
    description = "Access GitHub repositories, users, and other data via the REST API"
    category = TemplateCategory.API
    dataset_type = "url"
    documentation_url = "https://docs.github.com/en/rest"
    tags = ["github", "git", "repository", "api"]
    
    parameters = [
        TemplateParameter(
            name="endpoint",
            description="API endpoint path (e.g., /repos/owner/repo)",
            param_type="string",
            required=True,
            example="/repos/octocat/hello-world",
        ),
        TemplateParameter(
            name="token",
            description="GitHub personal access token",
            param_type="string",
            required=False,
            secret=True,
        ),
        TemplateParameter(
            name="per_page",
            description="Results per page (max 100)",
            param_type="integer",
            required=False,
            default=30,
            min_value=1,
            max_value=100,
        ),
    ]
    
    examples = [
        {
            "description": "Get repository info",
            "params": {"endpoint": "/repos/microsoft/vscode"},
        },
        {
            "description": "List user repositories",
            "params": {"endpoint": "/users/octocat/repos", "per_page": 10},
        },
    ]
    
    def create_dataset(self, **params) -> URLDataset:
        endpoint = params["endpoint"].lstrip("/")
        url = f"https://api.github.com/{endpoint}"
        
        # Add pagination params
        url_params = {}
        if params.get("per_page"):
            url_params["per_page"] = params["per_page"]
        
        # Build auth config
        auth = None
        if params.get("token"):
            auth = {"type": "bearer", "token": params["token"]}
        
        return URLDataset(
            url=url,
            content_type="json",
            params=url_params,
            auth=auth,
            headers={"Accept": "application/vnd.github.v3+json"},
        )
