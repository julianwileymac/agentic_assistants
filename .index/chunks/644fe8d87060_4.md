    def create_dataset(self, **params) -> AbstractDataset:
        """Create dataset from generic configuration."""
        from agentic_assistants.data.datasets import (
            URLDataset,
            WebsiteDataset,
            RSSDataset,
            KaggleDataset,
            S3PublicDataset,
            ScraperDataset,
            APIDataset,
        )
        
        # Map dataset types to classes
        type_map = {
            "url": URLDataset,
            "website": WebsiteDataset,
            "rss": RSSDataset,
            "kaggle": KaggleDataset,
            "s3_public": S3PublicDataset,
            "scraper": ScraperDataset,
            "api": APIDataset,
        }
        
        dataset_class = type_map.get(self.dataset_type)
        if not dataset_class:
            raise ValueError(f"Unknown dataset type: {self.dataset_type}")
        
        # Build dataset config
        config = dict(self.dataset_config)
        
        # Substitute URL template
        if self.url_template:
            url = self.url_template.format(**params)
            config["url"] = url
        
        # Merge params into config
        for key, value in params.items():
            if key not in config:
                config[key] = value
        
        return dataset_class(**config)
