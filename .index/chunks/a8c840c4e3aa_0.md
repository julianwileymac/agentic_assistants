"""
Built-in data source templates.

This module provides pre-configured templates for common data sources
including APIs, datasets, and web services.
"""

from typing import List

from agentic_assistants.data.templates.base import DataSourceTemplate
from agentic_assistants.data.templates.builtin.api_templates import get_api_templates
from agentic_assistants.data.templates.builtin.dataset_templates import get_dataset_templates
from agentic_assistants.data.templates.builtin.web_templates import get_web_templates


def get_builtin_templates() -> List[DataSourceTemplate]:
    """Get all built-in templates."""
    templates = []
    templates.extend(get_api_templates())
    templates.extend(get_dataset_templates())
    templates.extend(get_web_templates())
    return templates


__all__ = ["get_builtin_templates"]
