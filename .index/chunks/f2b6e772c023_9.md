    def _resolve_dataset(self, name: str) -> Optional[DatasetConfig]:
        """
        Resolve a dataset name to its configuration.
        
        First checks explicit datasets, then tries pattern matching.
        
        Args:
            name: Dataset name
            
        Returns:
            DatasetConfig or None if not found
        """
        # Check explicit datasets first
        if name in self._datasets:
            return self._datasets[name]
        
        # Try pattern matching
        for pattern in self._patterns:
            if pattern.matches(name):
                config = pattern.create_config(name)
                # Cache the resolved config
                self._datasets[name] = config
                return config
        
        return None
    
    def _get_dataset_instance(self, name: str) -> AbstractDataset:
        """
        Get or create a dataset instance.
        
        Args:
            name: Dataset name
            
        Returns:
            Dataset instance
            
        Raises:
            DatasetError: If dataset not found or cannot be created
        """
        # Check cache
        if name in self._dataset_instances:
            return self._dataset_instances[name]
        
        # Resolve configuration
        config = self._resolve_dataset(name)
        if not config:
            raise DatasetError(f"Dataset not found in catalog: {name}")
        
        # Get dataset class
        dataset_class = get_dataset_type(config.type)
        if not dataset_class:
            raise DatasetError(f"Unknown dataset type: {config.type}")
        
        # Resolve credentials
        credentials = None
        if config.credentials:
            credentials = self._credentials.get(config.credentials)
        
        # Create instance
        try:
            instance = dataset_class(
                filepath=config.filepath,
                load_args=config.load_args,
                save_args=config.save_args,
                credentials=credentials,
                versioned=config.versioned,
            )
        except Exception as e:
            raise DatasetError(f"Failed to create dataset {name}: {e}") from e
        
        # Cache instance
        self._dataset_instances[name] = instance
        
        return instance
