    def discover_schema(self, source: Union[str, DataSource]) -> SchemaInfo:
        """
        Discover schema for a data source.
        
        Args:
            source: DataSource instance or source ID
            
        Returns:
            SchemaInfo with tables and columns
        """
        if isinstance(source, str):
            source_obj = self.get_source(source)
            if not source_obj:
                return SchemaInfo(
                    source_id=source,
                    source_type="unknown",
                    metadata={"error": "Source not found"},
                )
        else:
            source_obj = source
        
        handler = self.get_handler(source_obj.source_type)
        if not handler:
            return SchemaInfo(
                source_id=source_obj.id,
                source_type=source_obj.source_type,
                metadata={"error": f"No handler for: {source_obj.source_type}"},
            )
        
        config = source_obj.get_config()
        schema = handler.discover_schema(config)
        schema.source_id = source_obj.id
        
        return schema
