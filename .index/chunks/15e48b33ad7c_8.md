class MinIOStorageBackend:
    """
    MinIO/S3-compatible storage backend for shared artifacts.
    
    This backend stores artifacts in MinIO for distributed access
    across the Kubernetes cluster.
    
    Example:
        >>> backend = MinIOStorageBackend()
        >>> artifact_id = await backend.store_artifact(
        ...     data=b"file contents",
        ...     name="report.pdf",
        ...     tags=["reports", "2024"]
        ... )
    """
    
    def __init__(self, config: Optional[AgenticConfig] = None):
        self.config = config or AgenticConfig()
        self._storage = None
    
    @property
    def storage(self):
        """Get MinIO storage client."""
        if self._storage is None:
            from agentic_assistants.kubernetes.storage import MinIOStorage
            self._storage = MinIOStorage(config=self.config)
        return self._storage
    
    @property
    def bucket(self) -> str:
        """Get default bucket name."""
        return self.config.minio.default_bucket
    
    async def store_artifact(
        self,

        artifact_type: str = "generic",
        tags: Optional[list[str]] = None,

    ) -> str:
        """
        Store an artifact in MinIO.
        
        Args:
            data: Artifact data as bytes
            name: Artifact name
            artifact_type: Type for organization
            tags: Tags to apply
            metadata: Additional metadata
        
        Returns:
            Artifact ID
        """
        from uuid import uuid4
        from datetime import datetime
        
        artifact_id = str(uuid4())
        object_path = f"artifacts/{artifact_type}/{artifact_id}/{name}"
        
        # Build metadata
        meta = metadata or {}
        meta["artifact_id"] = artifact_id
        meta["artifact_type"] = artifact_type
        meta["original_name"] = name
        meta["stored_at"] = datetime.utcnow().isoformat()
        if tags:
            meta["tags"] = ",".join(tags)
        
        # Upload to MinIO
        success = await self.storage.upload_data(
            self.bucket,
            object_path,
            data,
            metadata=meta,
        )
        
        if not success:
            raise RuntimeError(f"Failed to store artifact: {name}")
        
        logger.info(f"Stored artifact in MinIO: {artifact_id}")
        return artifact_id
