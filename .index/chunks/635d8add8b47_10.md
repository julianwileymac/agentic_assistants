    async (url: string, { arg }: { arg?: { component_type?: string; save_component?: boolean } }) => {
      const params = new URLSearchParams();
      if (arg?.component_type) params.set('component_type', arg.component_type);
      if (arg?.save_component !== undefined) params.set('save_component', String(arg.save_component));
      const query = params.toString();

  );
