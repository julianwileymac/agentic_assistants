    """Template for fetching Wikipedia articles."""
    
    id = "wikipedia"

    dataset_type = "website"

    tags = ["wikipedia", "encyclopedia", "knowledge"]
    
    parameters = [
        TemplateParameter(
            name="title",
            description="Article title",
            param_type="string",
            required=True,
            example="Machine_learning",
        ),
        TemplateParameter(
            name="language",
            description="Wikipedia language code",
            param_type="string",
            required=False,
            default="en",
        ),
    ]
    
    def create_dataset(self, **params) -> WebsiteDataset:
        title = params["title"].replace(" ", "_")
        lang = params.get("language", "en")
        url = f"https://{lang}.wikipedia.org/wiki/{title}"
        
        return WebsiteDataset(
            url=url,
            selectors={
                "title": "h1#firstHeading",
                "content": "div.mw-parser-output",
                "categories": {"selector": "div#catlinks a", "multiple": True},
            },
        )
