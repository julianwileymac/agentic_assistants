"""
Distributed Caching Layer for the Agentic Framework.

This module provides a flexible caching system with multiple backends:
- Local LRU cache with disk spillover
- Redis for distributed caching
- Hybrid multi-tier caching (L1 local, L2 distributed)

Example:
    >>> from agentic_assistants.data.caching import get_cache, HybridCache
    >>> 
    >>> # Get configured cache
    >>> cache = get_cache()
    >>> 
    >>> # Store and retrieve
    >>> cache.set("key", {"data": "value"})
    >>> data = cache.get("key")
"""

from agentic_assistants.data.caching.base import (
    CacheBackend,
    CacheEntry,
    CacheConfig,
    CacheStats,
)
from agentic_assistants.data.caching.local_cache import LocalCache
from agentic_assistants.data.caching.redis_cache import RedisCache
from agentic_assistants.data.caching.hybrid_cache import HybridCache

# Global cache instance
_global_cache = None


def get_cache(config: CacheConfig = None) -> CacheBackend:
    """
    Get or create the global cache instance.
    
    Args:
        config: Cache configuration (uses defaults if None)
        
    Returns:
        Configured cache backend
    """
    global _global_cache
    
    if _global_cache is None:
        from agentic_assistants.config import AgenticConfig
        
        app_config = AgenticConfig()
        
        if config is None:
            config = CacheConfig()
        
        backend = config.backend
        
        if backend == "local":
            _global_cache = LocalCache(config)
        elif backend == "redis":
            _global_cache = RedisCache(config)
        elif backend == "hybrid":
            _global_cache = HybridCache(config)
        else:
            _global_cache = LocalCache(config)
    
    return _global_cache


def clear_cache() -> None:
    """Clear the global cache."""
    global _global_cache
    if _global_cache:
        _global_cache.clear()
