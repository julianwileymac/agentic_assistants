def _stop_tunnel(cmd: Any) -> None:

def _split_server(server: str) -> tuple[str, str, int]:
    if "@" in server:
        username, server = server.split("@", 1)
    else:
        username = getuser()
    if ":" in server:
        server, port_str = server.split(":")
        port = int(port_str)
    else:
        port = 22
    return username, server, port
