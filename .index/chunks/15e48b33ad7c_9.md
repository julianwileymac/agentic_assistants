    async def retrieve_artifact(
        self,
        artifact_id: str,
        artifact_type: str = "generic",
    ) -> Optional[bytes]:
        """
        Retrieve an artifact from MinIO.
        
        Args:
            artifact_id: Artifact ID
            artifact_type: Type for path resolution
        
        Returns:
            Artifact data or None if not found
        """
        # List objects to find the artifact
        prefix = f"artifacts/{artifact_type}/{artifact_id}/"
        objects = await self.storage.list_objects(
            self.bucket,
            prefix=prefix,
            recursive=True,
        )
        
        if not objects:
            return None
        
        # Get the first (and should be only) object
        object_path = objects[0]["name"]
        return await self.storage.download_data(self.bucket, object_path)
    
    async def delete_artifact(

    ) -> bool:
        """
        Delete an artifact from MinIO.
        
        Args:
            artifact_id: Artifact ID
            artifact_type: Type for path resolution
        
        Returns:
            True if deleted successfully
        """
        prefix = f"artifacts/{artifact_type}/{artifact_id}/"
        objects = await self.storage.list_objects(
            self.bucket,
            prefix=prefix,
            recursive=True,
        )
        
        success = True
        for obj in objects:
            if not await self.storage.delete_object(self.bucket, obj["name"]):
                success = False
        
        return success
