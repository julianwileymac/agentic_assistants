    def get_upstream(
        self,
        entity_id: str,
        max_depth: int = 10,
    ) -> List[DataRelation]:
        """
        Get upstream relations (sources) for an entity.
        
        Args:
            entity_id: Entity ID
            max_depth: Maximum traversal depth
        
        Returns:
            List of upstream relations
        """
        upstream = []
        visited = set()
        queue = [(entity_id, 0)]
        
        while queue:
            current_id, depth = queue.pop(0)
            
            if current_id in visited or depth > max_depth:
                continue
            
            visited.add(current_id)
            
            for relation in self._relations.values():
                if relation.target_id == current_id:
                    upstream.append(relation)
                    queue.append((relation.source_id, depth + 1))
        
        return upstream
