                pipeline_name=pipeline.name,

        try:
            # Connect to Dask cluster
            client = Client(scheduler)
            logger.info(f"Connected to Dask scheduler: {scheduler}")
            
            # Execute pipeline nodes
            outputs = {}
            node_results = {}
            
            for node in pipeline.nodes:
