        """

        Args:

        """

        if not path.exists():
            logger.warning(f"File not found: {path}")
            return None
        
        if not path.is_file():
            logger.warning(f"Not a file: {path}")
            return None
        
        # Check file size
        size = path.stat().st_size

            path=path.resolve(),
