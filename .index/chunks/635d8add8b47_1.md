export function useSystemHealth() {
  return useSWR<SystemHealth>(
    '/api/internal/health',
    async () => {
      const backendUrl = getBackendUrl();
      
      const checkService = async (url: string, name: string) => {
        try {
          const start = Date.now();
          const response = await fetch(`${url}/health`, { 
            method: 'GET',
            signal: AbortSignal.timeout(5000)
          });
          const latency = Date.now() - start;
          return {
            status: response.ok ? 'healthy' : 'unhealthy',
            url,
            latency_ms: latency,
          };
        } catch {
          return { status: 'unhealthy', url, message: `${name} not reachable` };
        }
      };

      const [backend, ollama, mlflow, jupyter] = await Promise.all([
        checkService(backendUrl, 'Backend'),
        checkService('http://localhost:11434', 'Ollama'),
        checkService(DEFAULT_MLFLOW_URL, 'MLFlow'),
        checkService(DEFAULT_JUPYTER_URL, 'JupyterLab'),
      ]);

      return {
        backend: backend as SystemHealth['backend'],
        ollama: ollama as SystemHealth['ollama'],
        mlflow: mlflow as SystemHealth['mlflow'],
        jupyter: jupyter as SystemHealth['jupyter'],
      };
    },
    { ...defaultSwrConfig, refreshInterval: 30000 }
  );
}

export function useProjects(params?: { page?: number; limit?: number; status?: string }) {
  const searchParams = new URLSearchParams();
  if (params?.page) searchParams.set('page', String(params.page));
  if (params?.limit) searchParams.set('limit', String(params.limit));
  if (params?.status) searchParams.set('status', params.status);
  const query = searchParams.toString();
  
  return useSWR<PaginatedResponse<Project>>(
    `/api/v1/projects${query ? `?${query}` : ''}`,
    fetcher,
    defaultSwrConfig
  );
}
