    def extract(self, soup, url: str) -> Dict[str, Any]:
        """Extract table data as list of dictionaries."""
        tables = soup.select(self.table_selector)
        
        if not tables or self.table_index >= len(tables):
            return {"rows": []}
        
        table = tables[self.table_index]
        rows = table.find_all("tr")
        
        if not rows:
            return {"rows": []}
        
        # Get headers
        header_row = rows[self.header_row]
        headers = [
            th.get_text(strip=True)
            for th in header_row.find_all(["th", "td"])
        ]
        
        # Parse data rows
        data_rows = []
        start_row = self.header_row + 1 + self.skip_rows
        
        for row in rows[start_row:]:
            cells = row.find_all(["td", "th"])
            if len(cells) == len(headers):
                row_data = {
                    headers[i]: cells[i].get_text(strip=True)
                    for i in range(len(headers))
                }
                data_rows.append(row_data)
        
        return {"rows": data_rows, "headers": headers}
