    parameters = [
        TemplateParameter(

            required=False,

        TemplateParameter(
            name="split",
            description="Dataset split",
            param_type="string",

            default="train",

    ]
    
    def create_dataset(self, **params) -> URLDataset:
        # Use Hugging Face datasets viewer API for preview
        dataset_name = params["dataset_name"]
        config = params.get("config", "default")
        split = params.get("split", "train")
        
        url = f"https://datasets-server.huggingface.co/rows?dataset={dataset_name}&config={config}&split={split}&offset=0&length=100"
        
        return URLDataset(
            url=url,
            content_type="json",
        )
