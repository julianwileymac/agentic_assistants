import pytest

from .. import jslink, jsdlink, ToggleButton

def test_jslink_args():
    with pytest.raises(TypeError):
        jslink()
    w1 = ToggleButton()
    with pytest.raises(TypeError):
        jslink((w1, 'value'))

    w2 = ToggleButton()
