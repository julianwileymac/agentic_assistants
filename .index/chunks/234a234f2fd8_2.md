interface Draft {
  type: 'project' | 'agent' | 'flow' | 'component';
  id?: string; // undefined for new items
  data: Partial<Project | Agent | Flow | Component>;
  isDirty: boolean;
  savedAt?: Date;
}

interface DraftStore {
  drafts: Map<string, Draft>;
  
  setDraft: (key: string, draft: Draft) => void;
  getDraft: (key: string) => Draft | undefined;
  removeDraft: (key: string) => void;
  markDirty: (key: string) => void;
  markClean: (key: string) => void;
  clearAllDrafts: () => void;
  hasDirtyDrafts: () => boolean;
}

export const useDraftStore = create<DraftStore>((set, get) => ({
  drafts: new Map(),

  setDraft: (key, draft) =>
    set((state) => {
      const newDrafts = new Map(state.drafts);
      newDrafts.set(key, draft);
      return { drafts: newDrafts };
    }),

  getDraft: (key) => get().drafts.get(key),

  removeDraft: (key) =>
    set((state) => {
      const newDrafts = new Map(state.drafts);
      newDrafts.delete(key);
      return { drafts: newDrafts };
    }),

  markDirty: (key) =>
    set((state) => {
      const draft = state.drafts.get(key);
      if (draft) {
        const newDrafts = new Map(state.drafts);
        newDrafts.set(key, { ...draft, isDirty: true });
        return { drafts: newDrafts };
      }
      return state;
    }),

  markClean: (key) =>
    set((state) => {
      const draft = state.drafts.get(key);
      if (draft) {
        const newDrafts = new Map(state.drafts);
        newDrafts.set(key, { ...draft, isDirty: false, savedAt: new Date() });
        return { drafts: newDrafts };
      }
      return state;
    }),

  clearAllDrafts: () => set({ drafts: new Map() }),

  hasDirtyDrafts: () => {
    const drafts = get().drafts;
    for (const draft of drafts.values()) {
      if (draft.isDirty) return true;
    }
    return false;
  },
}));

interface NotificationStore {
  notifications: Notification[];
  
  addNotification: (notification: Omit<Notification, 'id' | 'createdAt'>) => string;
  removeNotification: (id: string) => void;
  clearNotifications: () => void;
}

export const useNotificationStore = create<NotificationStore>((set) => ({
  notifications: [],

    const id = Math.random().toString(36).substring(7);

      ...notification,

    };
