def infer_layer_from_path(path: Union[str, Path]) -> Optional[DataLayer]:
    """
    Infer the data layer from a file path.
    
    Args:
        path: File path to analyze
        
    Returns:
        DataLayer or None if not matching convention
    """
    path_str = str(path)
    
    for layer in DataLayer:
        if layer.value in path_str:
            return layer
    
    return None


def suggest_next_layer(current_layer: DataLayer) -> Optional[DataLayer]:
    """
    Suggest the next layer in the data pipeline.
    
    Args:
        current_layer: Current data layer
        
    Returns:
        Next layer or None if at the end
    """
    layers = list(DataLayer)
    try:
        idx = layers.index(current_layer)
        if idx < len(layers) - 1:
            return layers[idx + 1]
    except ValueError:
        pass
    return None
