def is_importable(module, attr: str, only_modules) -> bool:
    if only_modules:
        try:
            mod = getattr(module, attr)
        except ModuleNotFoundError:
            # See gh-14434
            return False
        return inspect.ismodule(mod)
    else:
        return not(attr[:2] == '__' and attr[-2:] == '__')
