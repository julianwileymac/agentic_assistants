"""
Training data lineage tracking.

This module provides utilities for tracking the provenance and lineage
of data used in training LLMs.
"""

import json
import uuid
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from pathlib import Path
from typing import Any, Dict, List, Optional, Set

from agentic_assistants.utils.logging import get_logger

logger = get_logger(__name__)


class LineageRelationship(str, Enum):
    """Types of relationships between data entities."""
    
    DERIVED_FROM = "derived_from"       # Target was derived from source
    TRANSFORMS_TO = "transforms_to"     # Source transforms to target
    SUBSET_OF = "subset_of"             # Target is a subset of source
    MERGED_FROM = "merged_from"         # Target was merged from multiple sources
    FILTERED_FROM = "filtered_from"     # Target was filtered from source
    AUGMENTED_FROM = "augmented_from"   # Target was augmented from source
    TRAINED_ON = "trained_on"           # Model was trained on data
    EVALUATED_ON = "evaluated_on"       # Model was evaluated on data


@dataclass
class TrainingDataLineage:
    """
    Tracks the lineage of data used to train a model.
    
    This includes:
    - Source datasets used for training
    - Transformations applied to data
    - Quality metrics at time of training
    - Version information
    """
    
    id: str
    
    # Model reference
    model_id: str
    training_job_id: str
    
    # Dataset reference
    dataset_id: str
    dataset_version: str
    dataset_name: str
    
    # Lineage details
    transformation_steps: List[str] = field(default_factory=list)
    filters_applied: List[str] = field(default_factory=list)
    
    # Quality at training time
    quality_metrics: Dict[str, float] = field(default_factory=dict)
    sample_count: int = 0
    
    # Timing
    created_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())
    
    # Metadata
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    def to_dict(self) -> Dict[str, Any]:
        return {
            "id": self.id,
            "model_id": self.model_id,
            "training_job_id": self.training_job_id,
            "dataset_id": self.dataset_id,
            "dataset_version": self.dataset_version,
            "dataset_name": self.dataset_name,
            "transformation_steps": self.transformation_steps,
            "filters_applied": self.filters_applied,
            "quality_metrics": self.quality_metrics,
            "sample_count": self.sample_count,
            "created_at": self.created_at,
            "metadata": self.metadata,
        }
