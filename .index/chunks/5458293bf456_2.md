    def test_max_size(self):
        max_size = 3
        with mock.patch.object(peephole_opt, "MAX_SIZE", max_size):
            # optimized binary operation: size <= maximum size
            #
            # (9,) * size
            size = max_size
            result = (9,) * size
            code = Bytecode(
                [
                    Instr("LOAD_CONST", 9),
