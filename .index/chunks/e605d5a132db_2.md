    def __init__(
        self,
        data_root: Union[str, Path] = "data",
        custom_layers: Optional[Dict[str, LayerConfig]] = None,
        enforce_convention: bool = False,
    ):
        """
        Initialize the data layer manager.
        
        Args:
            data_root: Root directory for data storage
            custom_layers: Custom layer configurations
            enforce_convention: If True, validate operations against convention
        """
        self.data_root = Path(data_root)
        self.enforce_convention = enforce_convention
        
        # Initialize layer configs
        self.layers: Dict[DataLayer, LayerConfig] = dict(self.DEFAULT_LAYER_CONFIGS)
        
        # Apply custom layer configs
        if custom_layers:
            for layer_name, config in custom_layers.items():
                layer = DataLayer.from_string(layer_name)
                if layer:
                    self.layers[layer] = config
        
        logger.debug(f"DataLayerManager initialized with root: {self.data_root}")
    
    def get_layer_path(self, layer: Union[DataLayer, str]) -> Path:
        """
        Get the filesystem path for a layer.
        
        Args:
            layer: DataLayer enum or string name
            
        Returns:
            Path to the layer directory
        """
        if isinstance(layer, str):
            layer = DataLayer.from_string(layer)
            if not layer:
                raise ValueError(f"Unknown layer: {layer}")
        
        config = self.layers[layer]
        return self.data_root / config.path
    
    def get_layer_config(self, layer: Union[DataLayer, str]) -> LayerConfig:
        """Get the configuration for a layer."""
        if isinstance(layer, str):
            layer = DataLayer.from_string(layer)
            if not layer:
                raise ValueError(f"Unknown layer: {layer}")
        return self.layers[layer]
