# =============================================================================
# Agentic Assistants - Docker Compose Configuration
# =============================================================================
# This file defines services for the Agentic Assistants framework.
#
# Usage:
#   docker-compose up -d              # Start all services
#   docker-compose up -d mlflow       # Start only MLFlow
#   docker-compose up agentic-api     # Start only the API
#   docker-compose up agentic-ide     # Start only the IDE
#   docker-compose down               # Stop all services
#   docker-compose logs -f            # View logs
#
# Services:
#   - agentic-api: Python FastAPI backend (port 8080)
#   - agentic-ide: Theia IDE frontend (port 3000)
#   - MLFlow: Experiment tracking server (port 5000)
#   - Jaeger: Distributed tracing UI (port 16686)
#   - OTEL Collector: OpenTelemetry collector (port 4317)
# =============================================================================

services:
  # ===========================================================================
  # Agentic API - Python Backend
  # ===========================================================================
  agentic-api:
    build:
      context: .
      dockerfile: docker/Dockerfile.api
    container_name: agentic-api
    ports:
      - "8080:8080"
    volumes:
      - ./data:/app/data
      - ./mlruns:/app/mlruns
      - ./config:/app/config
    environment:
      - AGENTIC_DATA_DIR=/app/data
      - AGENTIC_LOG_LEVEL=INFO
      - MLFLOW_TRACKING_URI=http://mlflow:5000
      - OTEL_EXPORTER_OTLP_ENDPOINT=http://otel-collector:4317
    depends_on:
      - mlflow
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 3
    restart: unless-stopped

  # ===========================================================================
  # Agentic IDE (JupyterLab - primary)
  # ===========================================================================
  agentic-ide:
    build:
      context: .
      dockerfile: docker/Dockerfile.lab
    container_name: agentic-lab
    ports:
      - "3000:3000"
    volumes:
      - ./data:/workspace/data:cached
      - ./examples:/workspace/examples:cached
      - ./notebooks:/workspace/notebooks:cached
    environment:
      - MLFLOW_TRACKING_URI=http://mlflow:5000
    depends_on:
      - agentic-api
    restart: unless-stopped

  # ===========================================================================
  # Theia IDE (primary web IDE)
  # ===========================================================================
  theia-ide:
    build:
      context: ./frontend
      dockerfile: ../docker/Dockerfile.ide
    container_name: agentic-theia
    ports:
      - "3001:3001"
    volumes:
      - ./data:/home/theia/data:cached
      - ./examples:/home/theia/examples:cached
      - ./notebooks:/home/theia/notebooks:cached
      - ./config:/home/theia/config:cached
