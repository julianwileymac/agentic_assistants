class CSVURLTemplate(DataSourceTemplate):
    """Generic template for CSV files from URLs."""
    
    id = "csv-url"
    name = "CSV from URL"
    description = "Load a CSV file from any URL"
    category = TemplateCategory.DATASET
    dataset_type = "url"
    tags = ["csv", "url", "generic"]
    
    parameters = [
        TemplateParameter(
            name="url",
            description="URL to CSV file",
            param_type="string",
            required=True,
            pattern=r"^https?://",
        ),
        TemplateParameter(
            name="delimiter",
            description="CSV delimiter",
            param_type="string",
            required=False,
            default=",",
        ),
        TemplateParameter(
            name="encoding",
            description="File encoding",
            param_type="string",
            required=False,
            default="utf-8",
        ),
    ]
    
    def create_dataset(self, **params) -> URLDataset:
        return URLDataset(
            url=params["url"],
            content_type="csv",
            load_args={
                "delimiter": params.get("delimiter", ","),
                "encoding": params.get("encoding", "utf-8"),
            },
        )
