export function useKubernetesDeployments(params?: {
  namespace?: string;
  deployment_type?: string;
}) {
  const searchParams = new URLSearchParams();
  if (params?.namespace) searchParams.set('namespace', params.namespace);
  if (params?.deployment_type) searchParams.set('deployment_type', params.deployment_type);
  const query = searchParams.toString();
  
  return useSWR<K8sDeploymentInfo[]>(
    `/api/v1/kubernetes/deployments${query ? `?${query}` : ''}`,
    fetcher,
    { ...defaultSwrConfig, refreshInterval: 10000 }
  );
}
