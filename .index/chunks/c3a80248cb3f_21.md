        self, transport: asyncio.DatagramTransport, protocol: DatagramProtocol

        self._protocol = protocol

        self._send_guard = ResourceGuard("writing to")
        self._closed = False

    @property
    def _raw_socket(self) -> socket.socket:

    async def aclose(self) -> None:
