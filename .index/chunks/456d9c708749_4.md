    def get_info(self) -> Dict[str, Any]:
        """Get dataset metadata from Kaggle."""
        api = self._get_kaggle_api()
        
        if self.dataset_ref:
            # Get dataset info
            datasets = api.dataset_list(search=self.dataset_ref)
            for ds in datasets:
                if ds.ref == self.dataset_ref:
                    return {
                        "ref": ds.ref,
                        "title": ds.title,
                        "size": ds.size,
                        "last_updated": str(ds.lastUpdated),
                        "download_count": ds.downloadCount,
                        "vote_count": ds.voteCount,
                        "usability_rating": ds.usabilityRating,
                    }
            return {}
        else:
            # Get competition info
            competitions = api.competitions_list()
            for comp in competitions:
                if comp.ref == self.competition:
                    return {
                        "ref": comp.ref,
                        "title": comp.title,
                        "deadline": str(comp.deadline) if comp.deadline else None,
                        "category": comp.category,
                        "reward": comp.reward,
                        "team_count": comp.teamCount,
                    }
            return {}
