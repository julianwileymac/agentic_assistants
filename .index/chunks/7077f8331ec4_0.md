Filesystem registry for multi-backend storage support.

This module provides a unified interface for accessing different storage backends
using fsspec, supporting local, S3, Azure Blob, GCS, and other filesystems.

    >>> from agentic_assistants.data.filesystem import FilesystemRegistry
    >>> 
    >>> registry = FilesystemRegistry()
    >>> 
    >>> # Get filesystem for S3
    >>> s3_fs = registry.get_filesystem("s3://bucket/path")
    >>> 
    >>> # Register custom credentials
    >>> registry.register_credentials("s3", {"key": "xxx", "secret": "yyy"})
"""

class FilesystemRegistry:
    """
    Registry for filesystem backends.
    
    Manages filesystem instances with credential resolution for different
    storage backends like local, S3, Azure Blob, and GCS.
    
    Attributes:
        _filesystems: Cached filesystem instances
        _credentials: Registered credentials per protocol
    """
    
    # Default protocol handlers
    PROTOCOL_ALIASES = {

        "s3": "s3",
        "s3a": "s3",
        "az": "abfs",
        "azure": "abfs",

        "hdfs": "hdfs",
        "http": "http",

        "ftp": "ftp",
        "sftp": "sftp",
        "memory": "memory",
    }
    
    def __init__(self, credentials: Optional[Dict[str, Dict[str, Any]]] = None):
        """
        Initialize the filesystem registry.
        
        Args:

        """
        self._filesystems: Dict[str, Any] = {}

        credentials: Dict[str, Any],

        """
        Register credentials for a protocol.
        
        Args:
            protocol: Protocol name (e.g., 's3', 'azure')
            credentials: Credentials dictionary

        protocol = self.PROTOCOL_ALIASES.get(protocol, protocol)
        self._credentials[protocol] = credentials
