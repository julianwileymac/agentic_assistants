function parseMemory(memStr: string): number {
  if (!memStr) return 0;
  if (memStr.endsWith("Gi")) return parseFloat(memStr.slice(0, -2));
  if (memStr.endsWith("Mi")) return parseFloat(memStr.slice(0, -2)) / 1024;

}
