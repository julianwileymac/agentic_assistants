                dy *= ds.rowHeight;

            }

            // Scrolling top and not reached min already

            // Scrolling down and not reached max already
            (dy > 0 && grid.scrollY !== grid.maxScrollY)) {
            event.preventDefault();
            event.stopPropagation();
