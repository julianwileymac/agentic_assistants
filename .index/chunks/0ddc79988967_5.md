class ChainedAugmenter(Augmenter):
    """
    Chain multiple augmenters together.
    """
    
    name = "chained"
    
    def __init__(self, augmenters: List[Augmenter]):
        """
        Initialize chained augmenter.
        
        Args:
            augmenters: List of augmenters to apply in order
        """
        self.augmenters = augmenters
    
    def augment(self, chunk: Chunk, context: Dict[str, Any]) -> Chunk:
        for augmenter in self.augmenters:
            chunk = augmenter.augment(chunk, context)

def get_augmenter(name: str, **kwargs) -> Augmenter:
    """
    Get an augmenter by name.
    
    Args:
        name: Augmenter name
        **kwargs: Arguments for the augmenter
        
    Returns:
        Configured Augmenter instance
    """
    augmenters = {
        "metadata": MetadataAugmenter,
        "keywords": KeywordAugmenter,
        "summary": SummaryAugmenter,
        "entities": EntityAugmenter,
        "qa": QAGenerationAugmenter,
    }
    
    augmenter_class = augmenters.get(name.lower())
    if not augmenter_class:
        raise ValueError(f"Unknown augmenter: {name}")
    
    return augmenter_class(**kwargs)
