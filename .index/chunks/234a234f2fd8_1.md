// ============================================================================
// Selection Store (for multi-select operations)
// ============================================================================

interface SelectionStore {
  selectedProjects: Set<string>;
  selectedAgents: Set<string>;
  selectedFlows: Set<string>;
  selectedComponents: Set<string>;

  toggleProject: (id: string) => void;
  toggleAgent: (id: string) => void;
  toggleFlow: (id: string) => void;
  toggleComponent: (id: string) => void;
  
  selectAllProjects: (ids: string[]) => void;
  selectAllAgents: (ids: string[]) => void;
  selectAllFlows: (ids: string[]) => void;
  selectAllComponents: (ids: string[]) => void;
  
  clearSelection: () => void;
}

export const useSelectionStore = create<SelectionStore>((set) => ({
  selectedProjects: new Set(),
  selectedAgents: new Set(),
  selectedFlows: new Set(),
  selectedComponents: new Set(),

  toggleProject: (id) =>
    set((state) => {
      const newSet = new Set(state.selectedProjects);
      if (newSet.has(id)) {
        newSet.delete(id);
      } else {
        newSet.add(id);
      }
      return { selectedProjects: newSet };
    }),

  toggleAgent: (id) =>
    set((state) => {
      const newSet = new Set(state.selectedAgents);
      if (newSet.has(id)) {
        newSet.delete(id);
      } else {
        newSet.add(id);
      }
      return { selectedAgents: newSet };
    }),

  toggleFlow: (id) =>
    set((state) => {
      const newSet = new Set(state.selectedFlows);
      if (newSet.has(id)) {
        newSet.delete(id);
      } else {
        newSet.add(id);
      }
      return { selectedFlows: newSet };
    }),

  toggleComponent: (id) =>
    set((state) => {
      const newSet = new Set(state.selectedComponents);
      if (newSet.has(id)) {
        newSet.delete(id);
      } else {
        newSet.add(id);
      }
      return { selectedComponents: newSet };
    }),

  selectAllProjects: (ids) => set({ selectedProjects: new Set(ids) }),
  selectAllAgents: (ids) => set({ selectedAgents: new Set(ids) }),
  selectAllFlows: (ids) => set({ selectedFlows: new Set(ids) }),
  selectAllComponents: (ids) => set({ selectedComponents: new Set(ids) }),

  clearSelection: () =>
    set({
      selectedProjects: new Set(),
      selectedAgents: new Set(),
      selectedFlows: new Set(),
      selectedComponents: new Set(),
    }),
}));
