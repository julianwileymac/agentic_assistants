"""
Data Catalog System.

This module provides a Kedro-inspired data catalog with YAML configuration,
schema discovery, metadata indexing, dataset factories, and data lineage tracking.

Features:
- YAML-based catalog configuration
- Dataset type registry with factory patterns
- Dataset versioning with timestamp-based versions
- Multi-environment configuration support
- Data lineage tracking

Example:
    >>> from agentic_assistants.data.catalog import DataCatalog
    >>> 
    >>> # Load catalog from YAML
    >>> catalog = DataCatalog.from_yaml("conf/base/catalog.yaml")
    >>> 
    >>> # Load a dataset
    >>> df = catalog.load("users")
    >>> 
    >>> # Save with versioning
    >>> catalog.save("processed_users", df)
    >>> 
    >>> # Search the catalog
    >>> results = catalog.search("user", entity_type="table")
"""

# ============================================================================
# Dataset Type Registry
# ============================================================================

# Registry mapping type strings to dataset classes
_DATASET_TYPE_REGISTRY: Dict[str, Type[AbstractDataset]] = {}


def register_dataset_type(type_name: str, dataset_class: Type[AbstractDataset]) -> None:
    """
    Register a dataset type with the catalog.
    
    Args:
        type_name: Type name (e.g., 'pandas.CSVDataset')
        dataset_class: Dataset class
    """
    _DATASET_TYPE_REGISTRY[type_name] = dataset_class
    logger.debug(f"Registered dataset type: {type_name}")


def get_dataset_type(type_name: str) -> Optional[Type[AbstractDataset]]:
    """Get a dataset class by type name."""
    return _DATASET_TYPE_REGISTRY.get(type_name)


def list_dataset_types() -> List[str]:
    """List all registered dataset types."""
    return list(_DATASET_TYPE_REGISTRY.keys())


# Register built-in dataset types
