@dataclass
class DataSourceTemplate(ABC):
    """
    Abstract base class for data source templates.
    
    Templates define:
    - What parameters are needed to configure a data source
    - How to create a dataset instance from those parameters
    - Documentation and examples
    
    Example:
        >>> class GitHubAPITemplate(DataSourceTemplate):
        ...     id = "github-api"
        ...     name = "GitHub API"
        ...     description = "Access GitHub repositories and data"
        ...     category = TemplateCategory.API
        ...     
        ...     parameters = [
        ...         TemplateParameter("owner", "Repository owner"),
        ...         TemplateParameter("repo", "Repository name"),
        ...         TemplateParameter("token", "GitHub token", secret=True, required=False),
        ...     ]
        ...     
        ...     def create_dataset(self, **params):
        ...         url = f"https://api.github.com/repos/{params['owner']}/{params['repo']}"
        ...         return URLDataset(url=url, ...)
    """
    
    # Template identification
    id: str = ""
    name: str = ""
    description: str = ""
    category: TemplateCategory = TemplateCategory.CUSTOM
    
    # Template configuration
    parameters: List[TemplateParameter] = field(default_factory=list)
    dataset_type: Optional[str] = None  # Hint for what dataset type is created
    
    # Documentation
    documentation_url: Optional[str] = None
    examples: List[Dict[str, Any]] = field(default_factory=list)
    tags: List[str] = field(default_factory=list)
    
    # State
    _parameter_map: Dict[str, TemplateParameter] = field(default_factory=dict, repr=False)
    
    def __post_init__(self):
        """Build parameter lookup map."""
        self._parameter_map = {p.name: p for p in self.parameters}
