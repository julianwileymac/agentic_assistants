class AbstractDataset(ABC, Generic[T]):
    """
    Abstract base class for all datasets.
    
    All dataset implementations must inherit from this class and implement
    the _load() and _save() methods.
    
    Attributes:
        filepath: Path to the data file/resource
        credentials: Optional credentials dictionary
        fs_args: Arguments for filesystem initialization
        metadata: Dataset metadata
    """
    
    DEFAULT_LOAD_ARGS: Dict[str, Any] = {}
    DEFAULT_SAVE_ARGS: Dict[str, Any] = {}
    
    def __init__(
        self,
        filepath: Union[str, Path],
        load_args: Optional[Dict[str, Any]] = None,
        save_args: Optional[Dict[str, Any]] = None,
        credentials: Optional[Dict[str, Any]] = None,
        fs_args: Optional[Dict[str, Any]] = None,
        versioned: bool = False,
        version: Optional[Version] = None,
        metadata: Optional[DatasetMetadata] = None,
    ):
        """
        Initialize the dataset.
        
        Args:
            filepath: Path to the data file/resource
            load_args: Arguments passed to the load method
            save_args: Arguments passed to the save method
            credentials: Credentials for filesystem access
            fs_args: Arguments for filesystem initialization
            versioned: Whether to enable versioning
            version: Specific version to use
            metadata: Dataset metadata
        """
        self._filepath = str(filepath)
        self._load_args = {**self.DEFAULT_LOAD_ARGS, **(load_args or {})}
        self._save_args = {**self.DEFAULT_SAVE_ARGS, **(save_args or {})}
        self._credentials = credentials or {}
        self._fs_args = fs_args or {}
        self._versioned = versioned
        self._version = version
        self._metadata = metadata or DatasetMetadata()
        
        # Lazy-initialized filesystem
        self._fs = None
        
        # Protocol detection
        self._protocol = self._detect_protocol(self._filepath)
    
    @staticmethod
    def _detect_protocol(filepath: str) -> str:
        """Detect the filesystem protocol from the filepath."""
        if "://" in filepath:
            return filepath.split("://")[0]
        return "file"
    
    @property
    def filepath(self) -> str:
        """Get the filepath, resolving version if needed."""
        if not self._versioned:
            return self._filepath
