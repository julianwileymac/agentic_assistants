        artifact_path = Path(metadata.shared_path)
        
        # Create tag links
        for tag in metadata.tags:

    def list_artifacts(
        self,
        tags: Optional[list[str]] = None,
        groups: Optional[list[str]] = None,
        user_id: Optional[str] = None,
        session_id: Optional[str] = None,
    ) -> list[SharedArtifactMetadata]:
        """
        List shared artifacts with optional filtering.
        """
        results = []
        
        for metadata in self._index.values():
            # Apply filters
            if tags and not any(t in metadata.tags for t in tags):
                continue
            if groups and not any(g in metadata.groups for g in groups):
                continue
            if user_id and metadata.source_user_id != user_id:
                continue
            if session_id and metadata.source_session_id != session_id:
                continue
            
            results.append(metadata)
        
        return results
    
    def add_tags(self, artifact_id: str, tags: list[str]) -> bool:
        """Add tags to an artifact."""
        if artifact_id not in self._index:
            return False
        
        metadata = self._index[artifact_id]
        new_tags = [t for t in tags if t not in metadata.tags]
        metadata.tags.extend(new_tags)
        
        self._save_index()
        self._update_links(metadata)
        
        return True
