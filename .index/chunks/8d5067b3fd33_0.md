and better context understanding.
"""

class Annotator(ABC):
    """Abstract base class for content annotators."""
    
    name: str = "base"
    
    @abstractmethod
    def annotate(self, chunk: Chunk, context: Dict[str, Any]) -> Chunk:
        """
        Annotate a chunk with labels/tags.
        
        Args:
            chunk: The chunk to annotate
            context: Additional context
            
        Returns:
            Annotated chunk
        """
        pass
    
    def annotate_batch(
        self,
        chunks: List[Chunk],
        context: Dict[str, Any],
    ) -> List[Chunk]:
        """Annotate multiple chunks."""
        return [self.annotate(chunk, context) for chunk in chunks]

    ):
        """
        Initialize classification annotator.
