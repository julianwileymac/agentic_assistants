    def list_projects(self, status: Optional[str] = None, 
                     page: int = 1, limit: int = 50) -> tuple[List[Project], int]:

        offset = (page - 1) * limit
        
        with self._get_connection() as conn:
            # Count total

                total = conn.execute(
                    "SELECT COUNT(*) FROM projects WHERE status = ?", (status,)
