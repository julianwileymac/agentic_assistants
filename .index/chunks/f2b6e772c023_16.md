    def search(
        self,
        query: str,
        entry_type: Optional[str] = None,
        source_id: Optional[str] = None,
        tags: Optional[List[str]] = None,
        limit: int = 50,
    ) -> CatalogSearchResult:
        """
        Search the catalog.
        
        Args:
            query: Search query (searches name and description)
            entry_type: Filter by entry type (table, column)
            source_id: Filter by source
            tags: Filter by tags (any match)
            limit: Maximum results
            
        Returns:
            CatalogSearchResult with matching entries
        """
        query_lower = query.lower()
        results = []
        
        for entry in self._entries.values():
            # Apply filters
            if entry_type and entry.entry_type != entry_type:
                continue
            if source_id and entry.source_id != source_id:
                continue
            if tags and not any(t in entry.tags for t in tags):
                continue
            
            # Search in name and description
            if (query_lower in entry.name.lower() or 
                query_lower in entry.description.lower()):
                results.append(entry)
            
            if len(results) >= limit:
                break
        
        return CatalogSearchResult(
            entries=results[:limit],
            total=len(results),
            query=query,
            filters={
                "entry_type": entry_type,
                "source_id": source_id,
                "tags": tags,
            },
        )
