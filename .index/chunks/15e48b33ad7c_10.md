    async def get_artifact_url(
        self,
        artifact_id: str,
        artifact_type: str = "generic",
        expires_hours: int = 1,
    ) -> Optional[str]:
        """
        Get a presigned URL for an artifact.
        
        Args:
            artifact_id: Artifact ID
            artifact_type: Type for path resolution
            expires_hours: URL expiration time
        
        Returns:
            Presigned URL or None
        """
        prefix = f"artifacts/{artifact_type}/{artifact_id}/"
        objects = await self.storage.list_objects(
            self.bucket,
            prefix=prefix,
            recursive=True,
        )
        
        if not objects:
            return None
        
        object_path = objects[0]["name"]
        return await self.storage.get_presigned_url(
            self.bucket,
            object_path,
            expires_hours,
        )
    
    async def copy_to_local(
        self,
        artifact_id: str,
        local_path: Path,
        artifact_type: str = "generic",
    ) -> bool:
        """
        Copy an artifact from MinIO to local storage.
        
        Args:
            artifact_id: Artifact ID
            local_path: Local destination path
            artifact_type: Type for path resolution
        
        Returns:
            True if copied successfully
        """
        data = await self.retrieve_artifact(artifact_id, artifact_type)
        if data is None:
            return False
        
        local_path.parent.mkdir(parents=True, exist_ok=True)
        local_path.write_bytes(data)
        return True
    
    async def copy_from_local(
        self,
        local_path: Path,
        artifact_type: str = "generic",
        tags: Optional[list[str]] = None,
    ) -> str:
        """
        Copy a local file to MinIO.
        
        Args:
            local_path: Local file path
            artifact_type: Type for organization
            tags: Tags to apply
        
        Returns:
            Artifact ID
        """
        if not local_path.exists():
            raise FileNotFoundError(f"Local file not found: {local_path}")
        
        data = local_path.read_bytes()
        return await self.store_artifact(
            data=data,
            name=local_path.name,
            artifact_type=artifact_type,
            tags=tags,
        )
