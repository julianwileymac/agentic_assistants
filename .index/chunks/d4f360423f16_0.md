#!/usr/bin/env bash
# Simple JupyterLab launcher for Agentic Assistants

set -e

PORT=${JUPYTER_PORT:-8888}
PROJECT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"

echo "Starting JupyterLab from ${PROJECT_DIR}"
echo "Port: ${PORT}"

cd "$PROJECT_DIR"

mkdir -p notebooks data mlruns

nohup jupyter lab \
  --ip 127.0.0.1 \
  --port ${PORT} \
  --no-browser \
  --NotebookApp.token='' \
  --NotebookApp.password='' \
  --NotebookApp.notebook_dir="${PROJECT_DIR}" \
  > jupyterlab.log 2>&1 &

PID=$!
echo $PID > .jupyter.pid

echo "JupyterLab started (PID: $PID) at http://localhost:${PORT}/lab"
echo "Logs: ${PROJECT_DIR}/jupyterlab.log"
