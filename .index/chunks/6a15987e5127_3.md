def get_registry() -> TemplateRegistry:
    """Get the global template registry."""
    global _global_registry
    if _global_registry is None:
        _global_registry = TemplateRegistry()
        # Load built-in templates
        _register_builtin_templates(_global_registry)
    return _global_registry


def _register_builtin_templates(registry: TemplateRegistry) -> None:
    """Register built-in templates."""
    try:
        from agentic_assistants.data.templates.builtin import get_builtin_templates
        for template in get_builtin_templates():
            registry.register(template, overwrite=True)
    except ImportError:
        logger.debug("No built-in templates available")


def get_template(template_id: str) -> DataSourceTemplate:
    """Get a template from the global registry."""
    return get_registry().get_or_raise(template_id)


def list_templates(category: Optional[str] = None) -> List[str]:
    """List template IDs from the global registry."""
    registry = get_registry()
    if category:
        return registry.list_by_category(category)
    return registry.list_all()


def register_template(template: Union[DataSourceTemplate, Dict[str, Any]]) -> None:
    """Register a template with the global registry."""
    get_registry().register(template)
