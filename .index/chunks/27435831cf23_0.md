"""
Built-in templates for dataset sources (Kaggle, public S3, etc.).
"""

from typing import List

from agentic_assistants.data.datasets import KaggleDataset, S3PublicDataset, URLDataset
from agentic_assistants.data.templates.base import (
    DataSourceTemplate,
    TemplateParameter,
    TemplateCategory,
)


class KaggleDatasetTemplate(DataSourceTemplate):
    """Template for Kaggle datasets."""
    
    id = "kaggle-dataset"
    name = "Kaggle Dataset"
    description = "Download datasets from Kaggle"
    category = TemplateCategory.DATASET
    dataset_type = "kaggle"
    documentation_url = "https://www.kaggle.com/docs/api"
    tags = ["kaggle", "ml", "dataset", "competition"]
    
    parameters = [
        TemplateParameter(
            name="dataset_ref",
            description="Dataset reference (username/dataset-name)",
            param_type="string",
            required=True,
            pattern=r"^[\w-]+/[\w-]+$",
            example="zillow/zecon",
        ),
        TemplateParameter(
            name="file_name",
            description="Specific file to load (optional)",
            param_type="string",
            required=False,
        ),
        TemplateParameter(
            name="kaggle_username",
            description="Kaggle username",
            param_type="string",
            required=False,
            secret=True,
        ),
        TemplateParameter(
            name="kaggle_key",
            description="Kaggle API key",
            param_type="string",
            required=False,
            secret=True,
        ),
    ]
    
    examples = [
        {
            "description": "Zillow housing dataset",
            "params": {"dataset_ref": "zillow/zecon"},
        },
    ]
    
    def create_dataset(self, **params) -> KaggleDataset:
        credentials = None
        if params.get("kaggle_username") and params.get("kaggle_key"):
            credentials = {
                "username": params["kaggle_username"],
                "key": params["kaggle_key"],
            }
        
        return KaggleDataset(
            dataset_ref=params["dataset_ref"],
            file_name=params.get("file_name"),
            credentials=credentials,
        )
